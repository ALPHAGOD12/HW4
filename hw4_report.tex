\section*{Question 1: Proof of the Eckart-Young Theorem (Exercise 7.2)}
%======================================================================

We prove the Eckart-Young theorem, which states that the best rank-$k$
approximation to a matrix in the Frobenius norm is given by its truncated SVD.

\subsection*{1.1 Theorem Statement}

\textbf{Theorem (Eckart-Young, 1936).}
Let $\boldsymbol{A} \in \mathbb{R}^{m \times n}$ with $\operatorname{rank}(\boldsymbol{A}) = r$
and singular value decomposition
\[
\boldsymbol{A} = \boldsymbol{U}\boldsymbol{\Sigma}\boldsymbol{V}^T
= \sum_{i=1}^{r} \sigma_i \,\boldsymbol{u}_i \boldsymbol{v}_i^T,
\]
where $\sigma_1 \geq \sigma_2 \geq \cdots \geq \sigma_r > 0$.
For any $k \leq r$, define
\[
\boldsymbol{A}_k = \sum_{i=1}^{k} \sigma_i \,\boldsymbol{u}_i \boldsymbol{v}_i^T
= \boldsymbol{U}_k \boldsymbol{\Sigma}_k \boldsymbol{V}_k^T,
\]
where $\boldsymbol{U}_k$, $\boldsymbol{\Sigma}_k$, $\boldsymbol{V}_k$ contain only
the first $k$ columns/entries. Then $\boldsymbol{A}_k$ solves
\[
\min_{\boldsymbol{B}:\;\operatorname{rank}(\boldsymbol{B})\leq k}
\|\boldsymbol{A} - \boldsymbol{B}\|_F^2,
\]
and the minimum value is $\sum_{i=k+1}^{r}\sigma_i^2$.

\subsection*{1.2 Proof}

The proof uses the \emph{unitary invariance} of the Frobenius norm:
for any orthogonal $\boldsymbol{P} \in \mathbb{R}^{m \times m}$ and
$\boldsymbol{Q} \in \mathbb{R}^{n \times n}$,
\[
\|\boldsymbol{P}\boldsymbol{M}\boldsymbol{Q}\|_F = \|\boldsymbol{M}\|_F.
\]

\paragraph{Step 1: Reduction via unitary invariance.}
Let $\boldsymbol{B}$ be any $m \times n$ matrix with $\operatorname{rank}(\boldsymbol{B}) \leq k$.
Since $\boldsymbol{A} = \boldsymbol{U}\boldsymbol{\Sigma}\boldsymbol{V}^T$, we write
\[
\|\boldsymbol{A} - \boldsymbol{B}\|_F
= \|\boldsymbol{U}\boldsymbol{\Sigma}\boldsymbol{V}^T - \boldsymbol{B}\|_F
= \|\boldsymbol{U}^T(\boldsymbol{U}\boldsymbol{\Sigma}\boldsymbol{V}^T - \boldsymbol{B})\boldsymbol{V}\|_F
= \|\boldsymbol{\Sigma} - \boldsymbol{U}^T\boldsymbol{B}\boldsymbol{V}\|_F.
\]
Define $\boldsymbol{N} := \boldsymbol{U}^T \boldsymbol{B}\boldsymbol{V}$.
Since $\boldsymbol{U}$ and $\boldsymbol{V}$ are orthogonal, $\operatorname{rank}(\boldsymbol{N}) = \operatorname{rank}(\boldsymbol{B}) \leq k$.
The problem reduces to
\[
\min_{\boldsymbol{N}:\;\operatorname{rank}(\boldsymbol{N})\leq k}
\|\boldsymbol{\Sigma} - \boldsymbol{N}\|_F^2.
\]

\paragraph{Step 2: Expand the Frobenius norm.}
Since $\boldsymbol{\Sigma}$ is an $m \times n$ matrix with $\Sigma_{ii} = \sigma_i$ for $i = 1,\dots,r$ and all other entries zero, we have
\begin{align}
\|\boldsymbol{\Sigma} - \boldsymbol{N}\|_F^2
&= \sum_{i=1}^{r}(\sigma_i - N_{ii})^2
 + \sum_{i=r+1}^{\min(m,n)} N_{ii}^2
 + \sum_{i \neq j} N_{ij}^2. \label{eq:expand}
\end{align}
Each of the three terms is a sum of squared real numbers and hence non-negative.

\paragraph{Step 3: Lower bound via diagonal restriction.}
Any $m \times n$ matrix $\boldsymbol{N}$ of rank at most $k$ has at most $k$ non-zero rows
(or more precisely, its diagonal can have at most $k$ non-zero entries if we want to simultaneously
zero out all off-diagonal terms). Observe:
\begin{itemize}
    \item The third term $\sum_{i\neq j}N_{ij}^2 \geq 0$ is minimised when $N_{ij}=0$ for all $i\neq j$
          (i.e., $\boldsymbol{N}$ is diagonal).
    \item The second term $\sum_{i=r+1}^{\min(m,n)} N_{ii}^2 \geq 0$ is minimised when $N_{ii}=0$ for $i > r$.
    \item Among diagonal $\boldsymbol{N}$ with rank $\leq k$, at most $k$ diagonal entries can be non-zero.
\end{itemize}
So the problem reduces to choosing at most $k$ of the indices $\{1,\dots,r\}$ at which to
set $N_{ii} = \sigma_i$, with $N_{ii} = 0$ at all remaining indices.

\paragraph{Step 4: Optimal index selection.}
Setting the off-diagonal terms and the terms for $i > r$ to zero, we need to minimise
\[
\sum_{i=1}^{r}(\sigma_i - N_{ii})^2
\quad\text{subject to at most $k$ of the $N_{ii}$ being nonzero.}
\]
For a given index $i$ where we \emph{do} set $N_{ii} \neq 0$, the unconstrained
minimiser of $(\sigma_i - N_{ii})^2$ is $N_{ii} = \sigma_i$ (reducing that squared term to zero).
For indices where $N_{ii} = 0$, the contribution is $\sigma_i^2$.
To minimise the total, we should zero out the $k$ \emph{largest} squared terms, i.e., set $N_{ii} = \sigma_i$ for $i = 1, 2, \dots, k$ (since $\sigma_1 \geq \sigma_2 \geq \cdots \geq \sigma_r$).

\paragraph{Step 5: Conclude.}
The optimal $\boldsymbol{N}^*$ is the diagonal matrix with entries
$N^*_{ii} = \sigma_i$ for $i=1,\dots,k$ and $N^*_{ii}=0$ otherwise.
Mapping back: $\boldsymbol{B}^* = \boldsymbol{U}\boldsymbol{N}^*\boldsymbol{V}^T = \sum_{i=1}^k \sigma_i\,\boldsymbol{u}_i\boldsymbol{v}_i^T = \boldsymbol{A}_k$.

The minimum approximation error is
\[
\|\boldsymbol{A} - \boldsymbol{A}_k\|_F^2
= \sum_{i=k+1}^{r}\sigma_i^2.
\]

\hfill$\blacksquare$

\subsection*{1.3 Remarks}

\begin{enumerate}
\item The elegance of this proof lies in the fact that unitary invariance of the Frobenius norm
completely decouples the optimisation into independent scalar problems on each entry
of the rotated matrix $\boldsymbol{N}$.

\item The same argument extends to the operator (spectral) norm:
$\min_{\operatorname{rank}(\boldsymbol{B})\leq k}\|\boldsymbol{A}-\boldsymbol{B}\| = \sigma_{k+1}$.
In that case, one uses the fact that the operator norm of $\boldsymbol{\Sigma}-\boldsymbol{N}$
is at least the absolute value of its largest diagonal entry that is not ``captured'' by $\boldsymbol{N}$.

\item The Eckart-Young theorem forms the theoretical foundation for PCA, low-rank matrix
completion, and many dimensionality reduction techniques.
\end{enumerate}

\section*{Question 2: PCA Derivation for Two Directions}
%======================================================================

We are given data $\boldsymbol{x}_1,\dots,\boldsymbol{x}_n \in \mathbb{R}^d$ with sample mean
$\bar{\boldsymbol{x}} = \frac{1}{n}\sum_{i=1}^n \boldsymbol{x}_i$ and centered data
$\boldsymbol{y}_i = \boldsymbol{x}_i - \bar{\boldsymbol{x}}$.
We seek an orthonormal matrix $\boldsymbol{U} = [\boldsymbol{u}_1\;\boldsymbol{u}_2] \in \mathbb{R}^{d \times 2}$
with $\boldsymbol{U}^T\boldsymbol{U} = \boldsymbol{I}_2$ that minimises the total squared
reconstruction error
\[
J(\boldsymbol{U})
= \sum_{i=1}^n \|\boldsymbol{y}_i - \hat{\boldsymbol{y}}_i\|^2,
\quad\text{where}\quad
\hat{\boldsymbol{y}}_i = \boldsymbol{U}\boldsymbol{U}^T\boldsymbol{y}_i.
\]

\subsection*{2.1 Reformulating the Objective}

Using the orthogonality of projection and its complement, we expand:
\begin{align}
\|\boldsymbol{y}_i - \boldsymbol{U}\boldsymbol{U}^T\boldsymbol{y}_i\|^2
&= \|(\boldsymbol{I} - \boldsymbol{U}\boldsymbol{U}^T)\boldsymbol{y}_i\|^2 \notag\\
&= \boldsymbol{y}_i^T(\boldsymbol{I} - \boldsymbol{U}\boldsymbol{U}^T)^T
   (\boldsymbol{I} - \boldsymbol{U}\boldsymbol{U}^T)\boldsymbol{y}_i \notag\\
&= \boldsymbol{y}_i^T(\boldsymbol{I} - \boldsymbol{U}\boldsymbol{U}^T)\boldsymbol{y}_i
\label{eq:idempotent}
\end{align}
since $\boldsymbol{I} - \boldsymbol{U}\boldsymbol{U}^T$ is an idempotent (projection) matrix
when $\boldsymbol{U}^T\boldsymbol{U} = \boldsymbol{I}_2$.

Summing over all data points:
\begin{align}
J(\boldsymbol{U})
&= \sum_{i=1}^n \boldsymbol{y}_i^T\boldsymbol{y}_i
 - \sum_{i=1}^n \boldsymbol{y}_i^T \boldsymbol{U}\boldsymbol{U}^T\boldsymbol{y}_i \notag\\
&= \underbrace{\sum_{i=1}^n \|\boldsymbol{y}_i\|^2}_{\text{constant}}
 - \sum_{i=1}^n \|\boldsymbol{U}^T\boldsymbol{y}_i\|^2.
\end{align}

Since the first term is a constant (independent of $\boldsymbol{U}$), minimising $J$ is
equivalent to \textbf{maximising}
\[
F(\boldsymbol{U}) := \sum_{i=1}^n \|\boldsymbol{U}^T\boldsymbol{y}_i\|^2.
\]

\subsection*{2.2 Expressing in Terms of the Covariance Matrix}

Define the $d \times d$ sample covariance matrix (up to scaling by $1/n$):
\[
\boldsymbol{S} := \sum_{i=1}^n \boldsymbol{y}_i\boldsymbol{y}_i^T.
\]
Now expand the objective. Since $\boldsymbol{U}^T\boldsymbol{y}_i \in \mathbb{R}^2$, we have
$\|\boldsymbol{U}^T\boldsymbol{y}_i\|^2 = \boldsymbol{y}_i^T\boldsymbol{U}\boldsymbol{U}^T\boldsymbol{y}_i$, so
\begin{align}
F(\boldsymbol{U})
&= \sum_{i=1}^n \boldsymbol{y}_i^T\boldsymbol{U}\boldsymbol{U}^T\boldsymbol{y}_i
= \operatorname{tr}\!\left(\boldsymbol{U}^T\!\left(\sum_{i=1}^n\boldsymbol{y}_i\boldsymbol{y}_i^T\right)\!\boldsymbol{U}\right)
= \operatorname{tr}(\boldsymbol{U}^T \boldsymbol{S}\,\boldsymbol{U}).
\end{align}

Here we used the trace identity $\boldsymbol{a}^T\boldsymbol{M}\boldsymbol{a} = \operatorname{tr}(\boldsymbol{a}\boldsymbol{a}^T\boldsymbol{M})$
and linearity of trace.

Writing $\boldsymbol{U} = [\boldsymbol{u}_1\;\boldsymbol{u}_2]$:
\[
\operatorname{tr}(\boldsymbol{U}^T\boldsymbol{S}\,\boldsymbol{U})
= \boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_1
+ \boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_2.
\]

\subsection*{2.3 Constrained Optimisation via Lagrange Multipliers}

We wish to maximise
\[
F(\boldsymbol{u}_1,\boldsymbol{u}_2)
= \boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_1
+ \boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_2
\]
subject to the three orthonormality constraints
$\boldsymbol{U}^T\boldsymbol{U} = \boldsymbol{I}_2$, i.e.,
\[
\boldsymbol{u}_1^T\boldsymbol{u}_1 = 1,\qquad
\boldsymbol{u}_2^T\boldsymbol{u}_2 = 1,\qquad
\boldsymbol{u}_1^T\boldsymbol{u}_2 = 0.
\]

Form the Lagrangian with multipliers $\lambda_1, \lambda_2$ (scalars) and $\mu$ (scalar):
\[
\mathcal{L} = \boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_1
+ \boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_2
- \lambda_1(\boldsymbol{u}_1^T\boldsymbol{u}_1 - 1)
- \lambda_2(\boldsymbol{u}_2^T\boldsymbol{u}_2 - 1)
- \mu\,\boldsymbol{u}_1^T\boldsymbol{u}_2.
\]

\paragraph{Differentiate with respect to $\boldsymbol{u}_1$:}
\[
\frac{\partial\mathcal{L}}{\partial\boldsymbol{u}_1}
= 2\boldsymbol{S}\,\boldsymbol{u}_1 - 2\lambda_1\boldsymbol{u}_1 - \mu\,\boldsymbol{u}_2
= \boldsymbol{0}.
\tag{I}
\]

\paragraph{Differentiate with respect to $\boldsymbol{u}_2$:}
\[
\frac{\partial\mathcal{L}}{\partial\boldsymbol{u}_2}
= 2\boldsymbol{S}\,\boldsymbol{u}_2 - 2\lambda_2\boldsymbol{u}_2 - \mu\,\boldsymbol{u}_1
= \boldsymbol{0}.
\tag{II}
\]

\subsection*{2.4 Showing $\mu = 0$ (the Cross-Multiplier Vanishes)}

Left-multiply equation (I) by $\boldsymbol{u}_2^T$:
\[
2\boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_1
- 2\lambda_1\underbrace{\boldsymbol{u}_2^T\boldsymbol{u}_1}_{=\,0}
- \mu\underbrace{\boldsymbol{u}_2^T\boldsymbol{u}_2}_{=\,1} = 0
\quad\Longrightarrow\quad
2\boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_1 = \mu.
\tag{III}
\]

Left-multiply equation (II) by $\boldsymbol{u}_1^T$:
\[
2\boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_2
- 2\lambda_2\underbrace{\boldsymbol{u}_1^T\boldsymbol{u}_2}_{=\,0}
- \mu\underbrace{\boldsymbol{u}_1^T\boldsymbol{u}_1}_{=\,1} = 0
\quad\Longrightarrow\quad
2\boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_2 = \mu.
\tag{IV}
\]

Left-multiply (I) by $\boldsymbol{u}_1^T$:
$2\boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_1
- 2\lambda_1 - \mu\underbrace{\boldsymbol{u}_1^T\boldsymbol{u}_2}_{=\,0} = 0$,
so $\boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_1 = \lambda_1$.
Similarly, from (II): $\boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_2 = \lambda_2$.

From (I) and (II), we can rewrite:
\begin{align}
\boldsymbol{S}\boldsymbol{u}_1 &= \lambda_1\boldsymbol{u}_1 + \tfrac{\mu}{2}\boldsymbol{u}_2, \label{eq:su1}\\
\boldsymbol{S}\boldsymbol{u}_2 &= \lambda_2\boldsymbol{u}_2 + \tfrac{\mu}{2}\boldsymbol{u}_1. \label{eq:su2}
\end{align}

This shows that $\operatorname{span}(\boldsymbol{u}_1,\boldsymbol{u}_2)$ is a 2-dimensional
invariant subspace of $\boldsymbol{S}$. In the basis $\{\boldsymbol{u}_1,\boldsymbol{u}_2\}$,
$\boldsymbol{S}$ restricted to this subspace has the $2\times 2$ representation
\[
\boldsymbol{S}\big|_{\operatorname{span}} =
\begin{pmatrix}\lambda_1 & \mu/2 \\ \mu/2 & \lambda_2\end{pmatrix}.
\]

\paragraph{Key WLOG argument.}
We can always find a $2\times 2$ orthogonal rotation $\boldsymbol{R}$ that diagonalises
this symmetric matrix, giving new orthonormal vectors
$[\boldsymbol{u}_1'\;\boldsymbol{u}_2'] = [\boldsymbol{u}_1\;\boldsymbol{u}_2]\boldsymbol{R}$
that are eigenvectors of $\boldsymbol{S}$ with eigenvalues $\tilde\lambda_1, \tilde\lambda_2$
(where $\tilde\lambda_1 + \tilde\lambda_2 = \lambda_1 + \lambda_2$ by trace preservation).
Crucially,
$\operatorname{span}(\boldsymbol{u}_1',\boldsymbol{u}_2') = \operatorname{span}(\boldsymbol{u}_1,\boldsymbol{u}_2)$,
and since the projection $\boldsymbol{U}\boldsymbol{U}^T$ depends only on the column space
(not the specific basis), the objective $J$ is unchanged.

Therefore, we may \textbf{without loss of generality} take $\mu = 0$, and the KKT conditions become:
\[
\boxed{\boldsymbol{S}\,\boldsymbol{u}_1 = \lambda_1\,\boldsymbol{u}_1
\qquad\text{and}\qquad
\boldsymbol{S}\,\boldsymbol{u}_2 = \lambda_2\,\boldsymbol{u}_2.}
\]
That is, $\boldsymbol{u}_1$ and $\boldsymbol{u}_2$ are \textbf{eigenvectors of $\boldsymbol{S}$}
with eigenvalues $\lambda_1$ and $\lambda_2$ respectively.

\subsection*{2.5 Identifying the Optimal Eigenvectors}

At the optimal solution:
\[
F^* = \operatorname{tr}(\boldsymbol{U}^T\boldsymbol{S}\,\boldsymbol{U})
= \boldsymbol{u}_1^T\boldsymbol{S}\,\boldsymbol{u}_1 + \boldsymbol{u}_2^T\boldsymbol{S}\,\boldsymbol{u}_2
= \lambda_1 + \lambda_2.
\]
To maximise $F^* = \lambda_1 + \lambda_2$, we must choose the eigenvectors corresponding
to the two \textbf{largest} eigenvalues of $\boldsymbol{S}$.

Let $\lambda^{(1)} \geq \lambda^{(2)} \geq \cdots \geq \lambda^{(d)} \geq 0$ be the eigenvalues
of $\boldsymbol{S}$ (all non-negative since $\boldsymbol{S}$ is positive semidefinite).
The optimal choice is:
\[
\boldsymbol{u}_1^* = \text{eigenvector of } \boldsymbol{S} \text{ for } \lambda^{(1)},\qquad
\boldsymbol{u}_2^* = \text{eigenvector of } \boldsymbol{S} \text{ for } \lambda^{(2)}.
\]

\subsection*{2.6 Final Result}

The minimum reconstruction error is
\[
J^* = \sum_{i=1}^n \|\boldsymbol{y}_i\|^2 - (\lambda^{(1)} + \lambda^{(2)})
= \operatorname{tr}(\boldsymbol{S}) - \lambda^{(1)} - \lambda^{(2)}
= \sum_{j=3}^{d}\lambda^{(j)}.
\]

\textbf{Conclusion:} The optimal 2-dimensional PCA subspace is spanned by the eigenvectors
of the sample scatter matrix $\boldsymbol{S} = \sum_{i=1}^n \boldsymbol{y}_i\boldsymbol{y}_i^T$
corresponding to its two largest eigenvalues. This naturally generalises the 1-direction
PCA derivation done in class: for one direction, the optimal $\boldsymbol{u}_1$ is the top
eigenvector; for two mutually perpendicular directions, the optimal
$\boldsymbol{U} = [\boldsymbol{u}_1\;\boldsymbol{u}_2]$ consists of the top two eigenvectors.

\section*{Question 3: Dictionary Conversion for Transformed Image Classes}
%======================================================================

\textbf{Setup.}
We have a learned dictionary $\boldsymbol{D} \in \mathbb{R}^{p \times K}$ (where $p$ is the
vectorised image dimension and $K$ is the number of atoms) such that every image
$\boldsymbol{x} \in \mathcal{S}$ admits a sparse representation
$\boldsymbol{x} \approx \boldsymbol{D}\boldsymbol{\alpha}$ with $\boldsymbol{\alpha}$ sparse.
The task is to derive a \emph{new} dictionary for each transformed class \textbf{without}
re-learning from scratch.

%----------------------------------------------------------------------
\subsection*{3(a) Class $\mathcal{S}_1$: Piecewise Affine Transform [5 points]}

Images in $\mathcal{S}_1$ are obtained by applying a known affine transform $\boldsymbol{A}_1$
to one subset and $\boldsymbol{A}_2$ to the other subset of images in $\mathcal{S}$.
The background is constant 0, and the transforms keep the foreground within the canvas.

\paragraph{Key observation.}
An affine spatial transformation on a vectorised image can be written as a
matrix--vector multiplication.
Let $\boldsymbol{T}_j \in \mathbb{R}^{p \times p}$ be the matrix that implements the
spatial affine transform $\boldsymbol{A}_j$ (via bilinear interpolation or
nearest-neighbour resampling) on a vectorised image.
Then for an image $\boldsymbol{x} \in \mathcal{S}$, the transformed image is
$\boldsymbol{x}' = \boldsymbol{T}_j\boldsymbol{x}$.

Since $\boldsymbol{x} = \boldsymbol{D}\boldsymbol{\alpha}$, we get
\[
\boldsymbol{x}' = \boldsymbol{T}_j\boldsymbol{D}\boldsymbol{\alpha}.
\]
The sparse code $\boldsymbol{\alpha}$ is \emph{the same}---only the dictionary changes.

\paragraph{New dictionary.}
Since $\mathcal{S}_1$ is the \emph{union} of images transformed by $\boldsymbol{A}_1$ and
$\boldsymbol{A}_2$, a single transformed dictionary may not suffice (an image transformed
by $\boldsymbol{A}_1$ will not be sparse under $\boldsymbol{T}_2\boldsymbol{D}$, in general).
The new dictionary is the \textbf{concatenation}:
\[
\boxed{\boldsymbol{D}_1 = [\boldsymbol{T}_1\boldsymbol{D} \;\;\; \boldsymbol{T}_2\boldsymbol{D}]
\;\in\;\mathbb{R}^{p \times 2K}.}
\]
Any image in $\mathcal{S}_1$ is sparse under $\boldsymbol{D}_1$: it uses atoms from
$\boldsymbol{T}_1\boldsymbol{D}$ or $\boldsymbol{T}_2\boldsymbol{D}$ (but not both),
so the coefficient vector has at most $s$ nonzeros in the relevant half and zeros in the
other half, preserving sparsity.

%----------------------------------------------------------------------
\subsection*{3(b) Class $\mathcal{S}_2$: Quadratic Intensity Transform [5 points]}

Images in $\mathcal{S}_2$ satisfy
$I_{\text{new}}(x,y) = \alpha\,[I_{\text{old}}(x,y)]^2 + \beta\,I_{\text{old}}(x,y) + \gamma$,
with known scalars $\alpha,\beta,\gamma$.
(We use bold $\boldsymbol{\alpha}$ for the sparse code and plain $\alpha$ for the transform parameter.)

\paragraph{Analysis.}
In vector form: $\boldsymbol{x}_{\text{new}} = \alpha\,\boldsymbol{x} \odot \boldsymbol{x} + \beta\,\boldsymbol{x} + \gamma\,\boldsymbol{1}$,
where $\odot$ denotes element-wise (Hadamard) product and $\boldsymbol{1}$ is the all-ones vector.

The $\beta\boldsymbol{x} + \gamma\boldsymbol{1}$ part is linear and can be handled easily:
$\beta\boldsymbol{D}\boldsymbol{\alpha} + \gamma\boldsymbol{1}$.
The problematic term is $\alpha(\boldsymbol{D}\boldsymbol{\alpha})\odot(\boldsymbol{D}\boldsymbol{\alpha})$,
which is \textbf{quadratic} in $\boldsymbol{\alpha}$---this breaks the linear
$\boldsymbol{x} = \boldsymbol{D}\boldsymbol{\alpha}$ model.

\paragraph{New dictionary.}
We cannot write $\boldsymbol{x}_{\text{new}} = \boldsymbol{D}_{\text{new}}\boldsymbol{\alpha}$
for the \emph{same} sparse $\boldsymbol{\alpha}$, because the squared term introduces
cross-products of atoms. Instead, we construct an \emph{expanded} dictionary.

Let $\boldsymbol{d}_j$ denote the $j$-th column (atom) of $\boldsymbol{D}$. If
$\boldsymbol{x} = \sum_j \alpha_j \boldsymbol{d}_j$, then
\[
\boldsymbol{x}\odot\boldsymbol{x} = \sum_{j}\sum_{k}\alpha_j\alpha_k\,(\boldsymbol{d}_j\odot\boldsymbol{d}_k).
\]
Define the ``lifted'' dictionary containing all pairwise Hadamard products:
\[
\boldsymbol{D}_{\text{quad}} = \{\boldsymbol{d}_j \odot \boldsymbol{d}_k : 1 \leq j \leq k \leq K\}
\;\in\;\mathbb{R}^{p \times \binom{K+1}{2}}.
\]
Then $\boldsymbol{x}_{\text{new}} = \alpha\,\boldsymbol{D}_{\text{quad}}\,\boldsymbol{c}
+ \beta\,\boldsymbol{D}\,\boldsymbol{\alpha} + \gamma\,\boldsymbol{1}$,
where $\boldsymbol{c}$ is the vector of products $(\alpha_j\alpha_k)_{j \leq k}$.
Writing this as a single dictionary--coefficient product:
\[
\boxed{\boldsymbol{D}_2 = [\alpha\,\boldsymbol{D}_{\text{quad}} \;\;\; \beta\,\boldsymbol{D} \;\;\; \gamma\,\boldsymbol{1}],
\quad
\boldsymbol{\alpha}_{\text{new}} = [\boldsymbol{c}^T,\;\boldsymbol{\alpha}^T,\;1]^T.}
\]
If $\boldsymbol{\alpha}$ has $s$ nonzeros, then $\boldsymbol{c}$ has at most $\binom{s+1}{2}$ nonzeros,
so $\boldsymbol{\alpha}_{\text{new}}$ has at most $\binom{s+1}{2} + s + 1$ nonzeros---still
sparse when $s \ll K$.

\paragraph{Special case $\alpha = 0$:} The transform becomes affine in intensity
($\boldsymbol{x}_{\text{new}} = \beta\boldsymbol{x} + \gamma\boldsymbol{1}$), and the new
dictionary is simply $\boldsymbol{D}_2 = [\beta\boldsymbol{D}\;\;\gamma\boldsymbol{1}]$.

%----------------------------------------------------------------------
\subsection*{3(c) Class $\mathcal{S}_4$: Downsampling by Factor $k$ [5 points]}

Images in $\mathcal{S}_4$ are obtained by downsampling by $k$ in both directions.

\paragraph{Key observation.}
Downsampling by $k$ in both X and Y is a \emph{linear} operation on the vectorised image.
Let $\boldsymbol{H} \in \mathbb{R}^{p' \times p}$ be the downsampling matrix (where
$p' = p/k^2$ for images of size $\sqrt{p}\times\sqrt{p}$; each row of $\boldsymbol{H}$
picks out the appropriate pixel or performs local averaging before subsampling).

If $\boldsymbol{x} = \boldsymbol{D}\boldsymbol{\alpha}$, then the downsampled image is
\[
\boldsymbol{x}_{\text{down}} = \boldsymbol{H}\boldsymbol{x} = \boldsymbol{H}\boldsymbol{D}\boldsymbol{\alpha}.
\]
The sparse code $\boldsymbol{\alpha}$ is preserved.

\paragraph{New dictionary.}
\[
\boxed{\boldsymbol{D}_4 = \boldsymbol{H}\boldsymbol{D} \;\in\;\mathbb{R}^{p' \times K}.}
\]
Each atom $\boldsymbol{d}_j$ is replaced by its downsampled version $\boldsymbol{H}\boldsymbol{d}_j$.
The number of atoms and the sparsity of $\boldsymbol{\alpha}$ are unchanged.

%----------------------------------------------------------------------
\subsection*{3(d) Class $\mathcal{S}_5$: Blur from a Known Set [5 points]}

Images in $\mathcal{S}_5$ are obtained by convolving images from $\mathcal{S}$ with a
blur kernel $\boldsymbol{h}$ that is a \emph{known} linear combination of kernels from
a known set $\mathcal{B} = \{\boldsymbol{b}_1, \dots, \boldsymbol{b}_L\}$:
\[
\boldsymbol{h} = \sum_{\ell=1}^L c_\ell\,\boldsymbol{b}_\ell,
\]
where the coefficients $c_\ell$ may vary from image to image (but the set $\mathcal{B}$ is known).

\paragraph{Key observation.}
Convolution is linear, so blurring a vectorised image $\boldsymbol{x}$ with kernel
$\boldsymbol{h}$ can be written as $\boldsymbol{B}_h\boldsymbol{x}$, where
$\boldsymbol{B}_h \in \mathbb{R}^{p \times p}$ is the convolution (Toeplitz/circulant) matrix
for $\boldsymbol{h}$. Since $\boldsymbol{h} = \sum_\ell c_\ell \boldsymbol{b}_\ell$,
linearity gives $\boldsymbol{B}_h = \sum_\ell c_\ell \boldsymbol{B}_\ell$, where
$\boldsymbol{B}_\ell$ is the convolution matrix for $\boldsymbol{b}_\ell$.

For $\boldsymbol{x} = \boldsymbol{D}\boldsymbol{\alpha}$:
\[
\boldsymbol{x}_{\text{blur}} = \boldsymbol{B}_h\boldsymbol{D}\boldsymbol{\alpha}
= \sum_{\ell=1}^L c_\ell\,\boldsymbol{B}_\ell\boldsymbol{D}\boldsymbol{\alpha}.
\]

Since the $c_\ell$ can vary per image, we cannot absorb them into the dictionary.
Instead, define $\boldsymbol{D}^{(\ell)} = \boldsymbol{B}_\ell\boldsymbol{D} \in \mathbb{R}^{p\times K}$
for each $\ell$. Then:
\[
\boldsymbol{x}_{\text{blur}}
= \sum_{\ell=1}^L c_\ell\,\boldsymbol{D}^{(\ell)}\boldsymbol{\alpha}
= [\boldsymbol{D}^{(1)} \;\cdots\; \boldsymbol{D}^{(L)}]
\begin{pmatrix} c_1\boldsymbol{\alpha} \\ \vdots \\ c_L\boldsymbol{\alpha} \end{pmatrix}.
\]

\paragraph{New dictionary.}
\[
\boxed{\boldsymbol{D}_5 = [\boldsymbol{B}_1\boldsymbol{D} \;\;\; \boldsymbol{B}_2\boldsymbol{D} \;\;\; \cdots \;\;\; \boldsymbol{B}_L\boldsymbol{D}]
\;\in\;\mathbb{R}^{p \times LK}.}
\]
The new coefficient vector is
$\tilde{\boldsymbol{\alpha}} = [c_1\boldsymbol{\alpha}^T,\dots,c_L\boldsymbol{\alpha}^T]^T \in \mathbb{R}^{LK}$.
If $\boldsymbol{\alpha}$ has $s$ nonzeros, then $\tilde{\boldsymbol{\alpha}}$ has at most $Ls$
nonzeros (the same support repeated $L$ times, scaled by different $c_\ell$), which is still
sparse when $s$ and $L$ are small.

%----------------------------------------------------------------------
\subsection*{3(e) Class $\mathcal{S}_6$: 2D Discrete Fourier Transform [5 points]}

Images in $\mathcal{S}_6$ are obtained by applying the 2D DFT to images in $\mathcal{S}$.

\paragraph{Key observation.}
The 2D DFT of a vectorised image $\boldsymbol{x}$ can be written as
$\boldsymbol{x}_{\text{DFT}} = \boldsymbol{F}\boldsymbol{x}$,
where $\boldsymbol{F} \in \mathbb{C}^{p \times p}$ is the unitary DFT matrix
(or equivalently $\boldsymbol{F} = \boldsymbol{F}_{n_1} \otimes \boldsymbol{F}_{n_2}$
for an $n_1 \times n_2$ image, where $\otimes$ is the Kronecker product and
$\boldsymbol{F}_{n}$ is the $n$-point DFT matrix).

Since $\boldsymbol{x} = \boldsymbol{D}\boldsymbol{\alpha}$:
\[
\boldsymbol{x}_{\text{DFT}} = \boldsymbol{F}\boldsymbol{x} = \boldsymbol{F}\boldsymbol{D}\boldsymbol{\alpha}.
\]
The sparse code $\boldsymbol{\alpha}$ is unchanged.

\paragraph{New dictionary.}
\[
\boxed{\boldsymbol{D}_6 = \boldsymbol{F}\boldsymbol{D} \;\in\;\mathbb{C}^{p \times K}.}
\]
Each atom $\boldsymbol{d}_j$ is replaced by its 2D DFT $\boldsymbol{F}\boldsymbol{d}_j$.
The number of atoms and the sparsity level are preserved exactly.
Note that $\boldsymbol{D}_6$ is in general complex-valued.

\section*{Question 4: LASSO Variant --- The Square-Root LASSO}
%======================================================================

\subsection*{4.1 Paper Details (2.5 points)}
\begin{itemize}
    \item \textbf{Title:} ``Square-Root Lasso: Pivotal Recovery of Sparse Signals via Conic Programming''
    \item \textbf{Authors:} Alexandre Belloni, Victor Chernozhukov, Lihong Wang
    \item \textbf{Venue:} \emph{Biometrika}, Volume 98, Issue 4
    \item \textbf{Year:} 2011
\end{itemize}

\subsection*{4.2 Mathematical Formulation (2.5 points)}
The standard LASSO solves
\[
\hat{\boldsymbol{\beta}}_{\text{LASSO}} = \arg\min_{\boldsymbol{\beta} \in \mathbb{R}^p}
\frac{1}{2n}\|\boldsymbol{y} - \boldsymbol{X}\boldsymbol{\beta}\|_2^2 + \lambda \|\boldsymbol{\beta}\|_1.
\]
The \textbf{Square-Root LASSO} replaces the squared loss with its square root:
\[
\boxed{\hat{\boldsymbol{\beta}}_{\text{SR}} = \arg\min_{\boldsymbol{\beta} \in \mathbb{R}^p}
\frac{1}{\sqrt{n}}\|\boldsymbol{y} - \boldsymbol{X}\boldsymbol{\beta}\|_2 + \lambda \|\boldsymbol{\beta}\|_1.}
\]

\paragraph{Notation and dimensions.}
\begin{itemize}
    \item $\boldsymbol{y} \in \mathbb{R}^n$: response vector ($n \times 1$).
    \item $\boldsymbol{X} \in \mathbb{R}^{n \times p}$: design matrix ($n$ observations, $p$ features).
    \item $\boldsymbol{\beta} \in \mathbb{R}^p$: coefficient vector ($p \times 1$).
    \item $\lambda > 0$: regularization parameter (scalar).
    \item $\|\cdot\|_2$: the Euclidean ($\ell_2$) norm.
    \item $\|\cdot\|_1$: the $\ell_1$ norm $\sum_{j=1}^p |\beta_j|$.
\end{itemize}

The model is $\boldsymbol{y} = \boldsymbol{X}\boldsymbol{\beta}^* + \boldsymbol{\varepsilon}$, where
$\boldsymbol{\varepsilon} \sim \mathcal{N}(\boldsymbol{0}, \sigma^2 \boldsymbol{I}_n)$ with \emph{unknown} $\sigma$.

\subsection*{4.3 Key Theorem (5 points)}

\begin{quote}
\textbf{Theorem (Belloni, Chernozhukov, Wang 2011).}
Suppose $\boldsymbol{y} = \boldsymbol{X}\boldsymbol{\beta}^* + \boldsymbol{\varepsilon}$ with $\boldsymbol{\varepsilon} \sim \mathcal{N}(\boldsymbol{0},\sigma^2\boldsymbol{I}_n)$.
Let $s = \|\boldsymbol{\beta}^*\|_0$ be the sparsity of the true coefficient vector. Choose the regularization parameter as
\[
\lambda = c \sqrt{\frac{\log p}{n}},
\]
where $c>0$ is a universal constant that does \emph{not} depend on $\sigma$.
Then, under a restricted eigenvalue condition on $\boldsymbol{X}$, the Square-Root LASSO estimator $\hat{\boldsymbol{\beta}}_{\text{SR}}$ satisfies
\[
\|\hat{\boldsymbol{\beta}}_{\text{SR}} - \boldsymbol{\beta}^*\|_2
\;\leq\; C\,\sigma\,\sqrt{\frac{s\,\log p}{n}},
\]
with high probability (at least $1 - O(1/p)$), where $C$ is a constant depending on the restricted eigenvalue of $\boldsymbol{X}$.
\end{quote}

The critical property is that $\lambda$ is \textbf{pivotal}: it does not depend on the unknown noise level $\sigma$. This is in contrast to the standard LASSO, where the optimal $\lambda$ scales as $\sigma\sqrt{\log p/n}$ and thus requires knowledge or estimation of $\sigma$.

\subsection*{4.4 Advantage over LASSO (10 points)}

The central advantage of the Square-Root LASSO over the standard LASSO is its
\textbf{pivotal} (self-normalizing) nature:

\begin{enumerate}
    \item \textbf{No need to know or estimate $\sigma$.}
    In the standard LASSO, the theoretically optimal regularization parameter is
    $\lambda_{\text{LASSO}} \propto \sigma\sqrt{\log p/n}$.
    Since the noise level $\sigma$ is almost never known in practice, one must either
    estimate it (which is itself a hard problem in the $p > n$ regime) or use
    computationally expensive cross-validation. The Square-Root LASSO removes this
    dependence entirely: $\lambda_{\text{SR}} \propto \sqrt{\log p/n}$, which depends only
    on the known quantities $n$ and $p$.

    \item \textbf{Equivalent near-oracle performance.}
    Despite the simpler tuning, the Square-Root LASSO achieves the same near-minimax
    optimal rate $O(\sigma\sqrt{s\log p/n})$ for estimation error as the optimally-tuned LASSO.
    There is no statistical price for the simpler calibration.

    \item \textbf{Practical robustness.}
    Because the estimator is pivotal, it is more robust to heteroscedasticity and
    model misspecification of the noise distribution. Even if $\sigma$ varies or is
    hard to estimate, the same $\lambda$ works.

    \item \textbf{Conic programming formulation.}
    The Square-Root LASSO can be reformulated as a second-order cone program (SOCP),
    which is a convex optimization problem solvable in polynomial time by standard
    interior-point methods. This makes it computationally tractable and allows the use
    of off-the-shelf solvers.
\end{enumerate}

In summary, the Square-Root LASSO retains all the theoretical guarantees of the LASSO
while eliminating the need for noise-level estimation, making it more practical and
easier to deploy in real-world high-dimensional settings.

\section*{Question 5: SVT for Low-Rank Matrix Completion}
%======================================================================

\subsection*{5.1 Problem Setup}

We solve the low-rank matrix completion problem via the Singular Value Thresholding
(SVT) algorithm.  Specifically, we use the Inexact Augmented Lagrange Multiplier (IALM)
variant, which converges significantly faster than vanilla SVT.

\paragraph{Matrix generation (Eckart-Young).}
A rank-$r$ matrix $\boldsymbol{X} \in \mathbb{R}^{n \times n}$ is created as
$\boldsymbol{X} = \boldsymbol{A}\boldsymbol{B}^T$ where
$\boldsymbol{A}, \boldsymbol{B} \in \mathbb{R}^{n \times r}$ have i.i.d.\
$\mathcal{N}(0,1)$ entries.

\paragraph{Observation model.}
We observe $m = f \cdot n^2$ entries uniformly at random and add i.i.d.\ Gaussian noise
with standard deviation $f_\sigma = 0.02 \times \text{mean}(|\text{observed entries}|)$.

\paragraph{Algorithm (IALM for nuclear norm minimization).}
We solve
\[
\min_{\boldsymbol{X}} \;\lambda\|\boldsymbol{X}\|_* \quad\text{s.t.}\quad
\mathcal{P}_\Omega(\boldsymbol{X}) = \boldsymbol{b},
\]
via the augmented Lagrangian iteration:
\begin{align}
\boldsymbol{T}^{(k)} &= \boldsymbol{X}^{(k)}\;\text{with}\;
T^{(k)}_{ij} \leftarrow b_{ij} + Y^{(k)}_{ij}/\mu_k \;\;\forall (i,j)\in\Omega,\\
\boldsymbol{X}^{(k+1)} &= \mathcal{D}_{\lambda/\mu_k}\!\left(\boldsymbol{T}^{(k)}\right)
\quad\text{(soft-threshold singular values)},\\
Y^{(k+1)}_{ij} &= Y^{(k)}_{ij} + \mu_k(b_{ij} - X^{(k+1)}_{ij}),\quad (i,j)\in\Omega,\\
\mu_{k+1} &= \min(\rho\,\mu_k,\; 10^7),
\end{align}
where $\mathcal{D}_\tau(\boldsymbol{M}) = \boldsymbol{U}\max(\boldsymbol{\Sigma}-\tau\boldsymbol{I},0)\boldsymbol{V}^T$
denotes the SVT operator, and $\rho = 1.15$ is the penalty growth factor.

\paragraph{Cross-validation.}
The regularization parameter $\lambda$ is selected by 2-fold cross-validation on the
observed entries, from the candidate set $\{0.01, 0.1, 1, 10, 50\}$.

\paragraph{RMSE metric.}
$\text{RMSE} = \|\hat{\boldsymbol{X}} - \boldsymbol{X}\|_F \;/\; \|\boldsymbol{X}\|_F$.

\subsection*{5.4 Discussion}

\begin{enumerate}
    \item \textbf{Effect of observation fraction $f$:}
    As $f$ increases (more entries observed), RMSE decreases significantly.
    For example, with $r=3$ at $n=1000$: RMSE drops from $0.277$ at $f=0.05$
    to $0.043$ at $f=0.15$.

    \item \textbf{Effect of rank $r$:}
    Higher rank matrices are harder to recover, especially when few entries are observed.
    At $f=0.05$, $r=80$ gives RMSE $\approx 1.0$ (essentially no recovery), while
    $r=3$ gives $0.277$. However, when $f$ is sufficiently large relative to $r$
    (the degrees of freedom of a rank-$r$ matrix is $r(2n-r)$), all ranks achieve
    similar low RMSE ($\sim 0.05$).

    \item \textbf{Information-theoretic limit:}
    A rank-$r$ matrix of size $n \times n$ has $r(2n-r)$ degrees of freedom.
    Recovery requires $m \gtrsim r(2n-r)$ observations. When $fn^2 < r(2n-r)$,
    recovery is fundamentally impossible, explaining the RMSE $\approx 1$ cases.

    \item \textbf{Execution time:}
    Time scales primarily with $n$ (due to SVD cost) and $k_{\text{svd}}$ (which
    tracks the rank). The IALM variant is efficient: $n=1000$ completes in $3$--$12$s
    and $n=3000$ in $28$--$70$s using randomised truncated SVD.
\end{enumerate}
